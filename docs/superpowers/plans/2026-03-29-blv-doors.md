> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement interactive sliding doors for indoor BLV maps, triggered by clicking/Enter via the EVT event system.

**Architecture:** Parse door data from DLV files and event_id from BLV face extras. Door faces spawn as individual entities (not batched) so vertices can animate. Clicking a clickable face dispatches its event_id through the existing EventQueue, which may contain a ChangeDoorState event that triggers the door animation state machine.

**Tech Stack:** Rust, Bevy 0.18 ECS, byteorder for binary parsing

**Files:**
- Modify: `openmm-data/src/dlv.rs` (rewrite DLV parser to read doors)

The door headers and data arrays live in the DLV (delta) file, not the BLV. The DLV sequential layout is:

1. LocationHeader (40 bytes)
2. visibleOutlines (875 bytes)
3. faceAttributes: u32 count + count × u32
4. decorationFlags: u32 count + count × u16
5. actors: u32 count + count × 548 bytes
6. spriteObjects: u32 count + count × variable (skip, count=0 typically)
7. chests: u32 count + count × variable (skip)
8. **doors: door_count × 80 bytes** (door_count from BLV)
9. **doorsData: (doors_data_size / 2) × i16** (doors_data_size from BLV header)
10. eventVariables, locationTime (skip)

```rust
    fn parse(data: &[u8], door_count: u32, doors_data_size: i32) -> Result<Self, Box<dyn Error>> {
        let mut cursor = Cursor::new(data);

        // 1. LocationHeader (40 bytes)
        cursor.seek(std::io::SeekFrom::Current(40))?;

        // 2. visibleOutlines (875 bytes)
        cursor.seek(std::io::SeekFrom::Current(875))?;

        // 3. faceAttributes: u32 count, then count × u32
        let face_attr_count = cursor.read_u32::<LittleEndian>()? as usize;
        cursor.seek(std::io::SeekFrom::Current(face_attr_count as i64 * 4))?;

        // 4. decorationFlags: u32 count, then count × u16
        let deco_flag_count = cursor.read_u32::<LittleEndian>()? as usize;
        cursor.seek(std::io::SeekFrom::Current(deco_flag_count as i64 * 2))?;

        // 5. actors: u32 count, then count × 548 bytes
        let actor_count = cursor.read_u32::<LittleEndian>()? as usize;
        let actors_start = cursor.position() as usize;
        let actors = Ddm::parse_actors_at(data, actors_start, actor_count);
        cursor.seek(std::io::SeekFrom::Current(actor_count as i64 * ACTOR_SIZE as i64))?;

        // 6. spriteObjects: u32 count, then count × 112 bytes
        let sprite_count = cursor.read_u32::<LittleEndian>()? as usize;
        cursor.seek(std::io::SeekFrom::Current(sprite_count as i64 * SPRITE_OBJECT_SIZE as i64))?;

        // 7. chests: u32 count, then count × 5076 bytes (MM6)
        let chest_count = cursor.read_u32::<LittleEndian>()? as usize;
        cursor.seek(std::io::SeekFrom::Current(chest_count as i64 * CHEST_SIZE_MM6 as i64))?;

        // 8. doors: door_count × 80 bytes (no count prefix — uses BLV's door_count)

        Self::unpack_door_data(&mut doors, &doors_data);
```

```rust
use std::error::Error;
use std::io::Cursor;

use byteorder::{LittleEndian, ReadBytesExt};

use crate::blv::{BlvDoor, DoorState};
use crate::ddm::{Ddm, DdmActor};
use crate::lod_data::LodData;
use crate::LodManager;

const SPRITE_OBJECT_SIZE: i64 = 112;
const CHEST_SIZE_MM6: i64 = 5076;
const ACTOR_SIZE: i64 = 548;

/// Parsed DLV indoor delta file.
pub struct Dlv {
    pub actors: Vec<DdmActor>,
    pub doors: Vec<BlvDoor>,
}

impl Dlv {
    pub fn new(
        lod_manager: &LodManager,
        map_name: &str,
        door_count: u32,
        doors_data_size: i32,
    ) -> Result<Self, Box<dyn Error>> {
        let dlv_name = map_name
            .rsplit_once('.')
            .map(|(base, _)| format!("{}.dlv", base))
            .unwrap_or_else(|| format!("{}.dlv", map_name));

        let raw = lod_manager.try_get_bytes(&format!("games/{}", dlv_name))?;
        let data = LodData::try_from(raw)?;
        Self::parse(&data.data, door_count, doors_data_size)
    }

    fn parse(data: &[u8], door_count: u32, doors_data_size: i32) -> Result<Self, Box<dyn Error>> {
        let mut cursor = Cursor::new(data);

        // 1. LocationHeader (40 bytes) + visibleOutlines (875 bytes)
        std::io::Seek::seek(&mut cursor, std::io::SeekFrom::Current(40 + 875))?;

        // 2. faceAttributes: u32 count + count × u32
        let count = cursor.read_u32::<LittleEndian>()? as i64;
        std::io::Seek::seek(&mut cursor, std::io::SeekFrom::Current(count * 4))?;

        // 3. decorationFlags: u32 count + count × u16
        let count = cursor.read_u32::<LittleEndian>()? as i64;
        std::io::Seek::seek(&mut cursor, std::io::SeekFrom::Current(count * 2))?;

        // 4. actors: u32 count + count × 548 bytes
        let actor_count = cursor.read_u32::<LittleEndian>()? as usize;
        let actors_start = cursor.position() as usize;
        let actors = Ddm::parse_actors_at(data, actors_start, actor_count);
        std::io::Seek::seek(&mut cursor, std::io::SeekFrom::Current(actor_count as i64 * ACTOR_SIZE))?;

        // 5. spriteObjects: u32 count + count × 112 bytes
        let count = cursor.read_u32::<LittleEndian>()? as i64;
        std::io::Seek::seek(&mut cursor, std::io::SeekFrom::Current(count * SPRITE_OBJECT_SIZE))?;

        // 6. chests: u32 count + count × 5076 bytes
        let count = cursor.read_u32::<LittleEndian>()? as i64;
        std::io::Seek::seek(&mut cursor, std::io::SeekFrom::Current(count * CHEST_SIZE_MM6))?;

        // 7. doors: door_count × 80 bytes (count from BLV, no prefix)
        let mut doors = Vec::with_capacity(door_count as usize);
        let mut door_counts = Vec::with_capacity(door_count as usize);
        for _ in 0..door_count {
            let (door, counts) = Self::read_door_header(&mut cursor)?;
            doors.push(door);
            door_counts.push(counts);
        }

        // 8. doorsData: (doors_data_size / 2) × i16
        let data_count = (doors_data_size / 2) as usize;
        let mut doors_data = Vec::with_capacity(data_count);
        for _ in 0..data_count {
            doors_data.push(cursor.read_i16::<LittleEndian>()?);
        }

        Self::unpack_door_data(&mut doors, &doors_data, &door_counts);

        Ok(Dlv { actors, doors })
    }

    /// Read an 80-byte door header. Returns the door and (num_vertices, num_faces, num_sectors, num_offsets).
    fn read_door_header(cursor: &mut Cursor<&[u8]>) -> Result<(BlvDoor, (u16, u16, u16, u16)), Box<dyn Error>> {
        let attributes = cursor.read_u32::<LittleEndian>()?;
        let door_id = cursor.read_u32::<LittleEndian>()?;
        let _time_since_triggered = cursor.read_u32::<LittleEndian>()?;

        // Direction: Vec3i fixed-point 16.16

        let dir_y = cursor.read_i32::<LittleEndian>()? as f32 / 65536.0;

        let open_speed = cursor.read_i32::<LittleEndian>()?;
        let close_speed = cursor.read_i32::<LittleEndian>()?;

        std::io::Seek::seek(cursor, std::io::SeekFrom::Current(32))?;

        let num_faces = cursor.read_u16::<LittleEndian>()?;

        let state = match state_raw {
            0 => DoorState::Open,
            1 => DoorState::Closing,
            3 => DoorState::Opening,
            _ => DoorState::Closed,
        };

        let door = BlvDoor {

            door_id,

            z_offsets: Vec::new(),

    /// Partition the flat doorsData blob into per-door arrays.
    fn unpack_door_data(
        doors: &mut [BlvDoor],
        data: &[i16],
        counts: &[(u16, u16, u16, u16)],
    ) {
        let mut j = 0usize;
        for (door, &(nv, nf, ns, no)) in doors.iter_mut().zip(counts.iter()) {
            let nv = nv as usize;
            let nf = nf as usize;
            let ns = ns as usize;
            let no = no as usize;

            if j + nv + nf + ns + nf + nf + no + no + no > data.len() {
                break;
            }

            door.vertex_ids = data[j..j + nv].iter().map(|&x| x as u16).collect();
            j += nv;

            door.face_ids = data[j..j + nf].iter().map(|&x| x as u16).collect();
            j += nf;

            // sector_ids — skip
            j += ns;

            door.delta_us = data[j..j + nf].to_vec();
            j += nf;

            door.delta_vs = data[j..j + nf].to_vec();
            j += nf;

            door.x_offsets = data[j..j + no].to_vec();
            j += no;

            door.y_offsets = data[j..j + no].to_vec();
            j += no;

            door.z_offsets = data[j..j + no].to_vec();
            j += no;
        }
    }
```

```rust
    pub fn door_face_meshes(
        &self,
        doors: &[BlvDoor],
        texture_sizes: &HashMap<String, (u32, u32)>,
    ) -> Vec<BlvDoorFaceMesh> {
        let mut result = Vec::new();
        for (door_idx, door) in doors.iter().enumerate() {
            for &face_id in &door.face_ids {
                let face_idx = face_id as usize;
                if face_idx >= self.faces.len() { continue; }
                let face = &self.faces[face_idx];
                if face.num_vertices < 3 || face.is_invisible() { continue; }

                let tex_name = if face_idx < self.texture_names.len() {
                    &self.texture_names[face_idx]
                } else { continue };
                if tex_name.is_empty() { continue; }

                let (tex_w, tex_h) = texture_sizes
                    .get(tex_name).copied().unwrap_or((128, 128));
                let tex_w_f = tex_w as f32;
                let tex_h_f = tex_h as f32;

                let mm6_normal = face.normal_f32();
                let normal = [mm6_normal[0], mm6_normal[2], -mm6_normal[1]];

                let mut positions = Vec::new();
                let mut normals = Vec::new();
                let mut uvs = Vec::new();

                let triangles = Self::triangulate_face(face, &self.vertices);
                for tri in &triangles {
                    for &vi in tri {
                        if vi < face.vertex_ids.len() {
                            let vert_idx = face.vertex_ids[vi] as usize;
                            if vert_idx < self.vertices.len() {
                                let v = &self.vertices[vert_idx];
                                positions.push(crate::odm::mm6_to_bevy(
                                    v.x as i32, v.y as i32, v.z as i32,
                                ));
                            } else {
                                positions.push([0.0, 0.0, 0.0]);
                            }
                        } else {
                            positions.push([0.0, 0.0, 0.0]);
                        }

                        let u = if vi < face.texture_us.len() {
                            (face.texture_us[vi] as f32 + face.texture_delta_u as f32) / tex_w_f
                        } else { 0.0 };
                        let v = if vi < face.texture_vs.len() {
                            (face.texture_vs[vi] as f32 + face.texture_delta_v as f32) / tex_h_f
                        } else { 0.0 };
                        uvs.push([u, v]);
                        normals.push(normal);
                    }
                }

                result.push(BlvDoorFaceMesh {
                    face_index: face_idx,
                    door_index: door_idx,
                    texture_name: tex_name.clone(),
                    positions,
                    normals,
                    uvs,
                });
            }
        }
        result
    }
```

```rust
/// Ray-plane intersection + point-in-polygon test for indoor face clicking.
fn indoor_interact_system(
    keys: Res<ButtonInput<KeyCode>>,
    mouse: Res<ButtonInput<MouseButton>>,
    gamepads: Query<&Gamepad>,
    camera_query: Query<(&GlobalTransform, &Camera), With<crate::game::player::PlayerCamera>>,
    clickable: Option<Res<ClickableFaces>>,
    door_faces: Query<(&DoorFace, &GlobalTransform, &Mesh3d)>,
    map_events: Option<Res<crate::game::events::MapEvents>>,
    mut event_queue: ResMut<crate::game::event_dispatch::EventQueue>,
    blv_doors: Option<Res<BlvDoors>>,
    cursor_query: Query<&bevy::window::CursorOptions, With<bevy::window::PrimaryWindow>>,
) {
    let (key, click, gamepad) = crate::game::interaction::check_interact_input(&keys, &mouse, &gamepads);
    if !key && !click && !gamepad { return; }

    let cursor_grabbed = cursor_query.single()
        .map(|c| !matches!(c.grab_mode, bevy::window::CursorGrabMode::None)).unwrap_or(true);
    if click && !cursor_grabbed { return; }

    let Ok((cam_global, _)) = camera_query.single() else { return };
    let ray_origin = cam_global.translation();
    let ray_dir = cam_global.forward().as_vec3();

    let Some(clickable) = clickable else { return };

    // Test against all clickable faces (including door faces)
    let mut best_hit: Option<(f32, u16)> = None; // (distance, event_id)

    for info in &clickable.faces {
        if let Some(t) = ray_plane_intersect(ray_origin, ray_dir, info.normal, info.plane_dist) {
            if t > 0.0 && t < 2000.0 {
                let hit_point = ray_origin + ray_dir * t;
                if point_in_polygon(hit_point, &info.vertices, info.normal) {
                    if best_hit.is_none() || t < best_hit.unwrap().0 {
                        best_hit = Some((t, info.event_id));
                    }
                }
            }
        }
    }

    if let Some((_, event_id)) = best_hit {
        let Some(me) = map_events else { return };
        let Some(evt) = me.evt.as_ref() else { return };
        event_queue.push_all(event_id, evt);
    }
}

/// Ray-plane intersection. Returns t (distance along ray) or None if parallel.
fn ray_plane_intersect(origin: Vec3, dir: Vec3, normal: Vec3, plane_dist: f32) -> Option<f32> {
    let denom = normal.dot(dir);
    if denom.abs() < 1e-6 { return None; }
    let t = (plane_dist - normal.dot(origin)) / denom;
    Some(t)
}

/// Test if a point lies inside a convex polygon (vertices in order).
fn point_in_polygon(point: Vec3, vertices: &[Vec3], normal: Vec3) -> bool {
    let n = vertices.len();
    if n < 3 { return false; }
    for i in 0..n {
        let a = vertices[i];
        let b = vertices[(i + 1) % n];
        let edge = b - a;
        let to_point = point - a;
        let cross = edge.cross(to_point);
        if cross.dot(normal) < 0.0 {
            return false;
        }
    }
    true
}
```

```rust
/// Trigger a door state change. action: 0=Open, 1=Close, 2=Toggle.
pub fn trigger_door(doors: &mut BlvDoors, door_id: u32, action: u8) {
    let Some(door) = doors.doors.iter_mut().find(|d| d.door_id == door_id) else {
        warn!("Door {} not found", door_id);
        return;
    };

    match action {
        0 => { // Open
            match door.state {
                DoorState::Closed => {
                    door.state = DoorState::Opening;
                    door.time_since_triggered_ms = 0.0;
                }
                DoorState::Closing => {
                    // Reverse: proportional time
                    let progress = door.time_since_triggered_ms * door.close_speed as f32 / 1000.0;
                    door.state = DoorState::Opening;
                    door.time_since_triggered_ms = (door.move_length as f32 - progress) * 1000.0 / door.open_speed as f32;
                }
                _ => {}
            }
        }
        1 => { // Close
            match door.state {
                DoorState::Open => {
                    door.state = DoorState::Closing;
                    door.time_since_triggered_ms = 0.0;
                }
                DoorState::Opening => {
                    let progress = door.move_length as f32 - door.time_since_triggered_ms * door.open_speed as f32 / 1000.0;
                    door.state = DoorState::Closing;
                    door.time_since_triggered_ms = progress * 1000.0 / door.close_speed as f32;
                }
                _ => {}
            }
        }
        2 | _ => { // Toggle
            match door.state {
                DoorState::Closed => {
                    door.state = DoorState::Opening;
                    door.time_since_triggered_ms = 0.0;
                }
                DoorState::Open => {
                    door.state = DoorState::Closing;
                    door.time_since_triggered_ms = 0.0;
                }
                DoorState::Closing => {
                    let progress = door.time_since_triggered_ms * door.close_speed as f32 / 1000.0;
                    door.state = DoorState::Opening;
                    door.time_since_triggered_ms = (door.move_length as f32 - progress) * 1000.0 / door.open_speed as f32;
                }
                DoorState::Opening => {
                    let progress = door.move_length as f32 - door.time_since_triggered_ms * door.open_speed as f32 / 1000.0;
                    door.state = DoorState::Closing;
                    door.time_since_triggered_ms = progress * 1000.0 / door.close_speed as f32;
```

```rust
fn door_animation_system(
    time: Res<Time>,
    mut blv_doors: Option<ResMut<BlvDoors>>,
    door_faces: Query<(&DoorFace, &Mesh3d)>,
    mut meshes: ResMut<Assets<Mesh>>,
) {
    let Some(ref mut doors_res) = blv_doors else { return };
    let dt_ms = time.delta_secs() * 1000.0;

    // Advance time and calculate distance for each door
    let mut distances: Vec<f32> = Vec::with_capacity(doors_res.doors.len());
    for door in doors_res.doors.iter_mut() {
        let distance = match door.state {
            DoorState::Open => 0.0,
            DoorState::Closed => door.move_length as f32,
            DoorState::Opening => {
                door.time_since_triggered_ms += dt_ms;
                let d = door.move_length as f32
                    - door.time_since_triggered_ms * door.open_speed as f32 / 1000.0;
                if d <= 0.0 {
                    door.state = DoorState::Open;
                    door.time_since_triggered_ms = 0.0;
                    0.0
                } else {
                    d
                }
            }
            DoorState::Closing => {
                door.time_since_triggered_ms += dt_ms;
                let d = door.time_since_triggered_ms * door.close_speed as f32 / 1000.0;
                if d >= door.move_length as f32 {
                    door.state = DoorState::Closed;
                    door.time_since_triggered_ms = 0.0;
                    door.move_length as f32
                } else {
                    d
                }
            }
        };
        distances.push(distance);
    }

    // Update mesh positions for door faces
    for (df, mesh_handle) in door_faces.iter() {
        let door = &doors_res.doors[df.door_index];

        // Skip static doors (not animating)
        if door.state == DoorState::Open || door.state == DoorState::Closed {
            continue;
        }

        let distance = distances[df.door_index];

        if let Some(mesh) = meshes.get_mut(&mesh_handle.0) {
            if let Some(positions) = mesh.attribute_mut(Mesh::ATTRIBUTE_POSITION) {
                let bevy::mesh::VertexAttributeValues::Float32x3(ref mut pos) = positions else { continue };

                for (i, p) in pos.iter_mut().enumerate() {
                    if i >= df.vertex_offset_indices.len() { break; }
                    let oi = df.vertex_offset_indices[i];
                    if oi >= door.x_offsets.len() { continue; }

                    // Compute position in MM6 coords: offset + direction * distance
                    let mm6_x = door.x_offsets[oi] as f32 + door.direction[0] * distance;
                    let mm6_y = door.y_offsets[oi] as f32 + door.direction[1] * distance;
                    let mm6_z = door.z_offsets[oi] as f32 + door.direction[2] * distance;

                    // Convert to Bevy coords
                    let bevy = openmm_data::odm::mm6_to_bevy(mm6_x as i32, mm6_y as i32, mm6_z as i32);
                    *p = bevy;
                }
            }
        }
    }
}
```
